```python
import pandas as pd
from pandas import Series, DataFrame
```
```python
obj = pd.Series([4, 7, -5, 3])
obj
0    4
1    7
2   -5
3    3
dtype: int64
```
```python
obj.values
array([ 4,  7, -5,  3])
```
```python
obj.index # like range(4)
RangeIndex(start=0, stop=4, step=1)
```
```python
obj2 = pd.Series([4, 7, -5, 3], index=['d', 'b', 'a', 'c'])
obj2
d    4
b    7
a   -5
c    3
dtype: int64
```
```python
obj2.index
Index(['d', 'b', 'a', 'c'], dtype='object')
```
```python
obj2['a']
-5
```
```python
obj2['d'] = 6
obj2[['c', 'a', 'd']]
c    3
a   -5
d    6
dtype: int64
```
